Public Sub Form_Open()
  Dim cHome As String
  Dim c_SysInfo1 As String
  Dim c_SysInfo2 As String
  Dim c_SysInfo3 As String
  Dim c_SysInfo4 As String
  Dim c_SysInfo5 As String
  Dim c_SysInfo6 As String
  Dim c_SysInfo7 As String
  Dim c_SysInfo8 As String
  Dim c_SysInfo9 As String
  Dim c_SysInfo10 As String
  Dim c_SysInfo11 As String
  Dim c_SysInfo12 As String
  Dim c_SysInfo13 As String
  Dim c_SysInfo14 As String
  Dim c_SysInfo15 As String
  Dim c_SysInfo16 As String
  Dim c_Temp As String
  Dim n_inici_GB_HDD As Integer
  Dim n_fi_GB_HDD As Integer
  ' System.language = "fr_FR.utf8"
  cHome = Application.Env["HOME"]
  ' RAM del PC on executem el programa
  ' RAM of the PC where the program is running
  Shell "cat /proc/meminfo | grep 'MemTotal'" To c_SysInfo1
  nRAM_MB = Val(Mid(c_SysInfo1, 17, 8)) / 1024
  TB_PC_RAM.Text = Trim(Str(nRAM_MB))
  ' Nom del processador
  ' Name of processor
  Shell "cat /proc/cpuinfo | grep 'model name'" To c_SysInfo2
  c_Temp = "model name"
  TA_PC_CPU.Text = Mid(c_SysInfo2, InStr(c_SysInfo2, c_Temp) + 13, 40)
  TA_PC_CPU.Text = Replace(TA_PC_CPU.Text, "(c)", "")
  TA_PC_CPU.Text = Replace(TA_PC_CPU.Text, "(C)", "")
  TA_PC_CPU.Text = Replace(TA_PC_CPU.Text, "(r)", "")
  TA_PC_CPU.Text = Replace(TA_PC_CPU.Text, "(R)", "")
  TA_PC_CPU.Text = Replace(TA_PC_CPU.Text, "(tm)", "")
  TA_PC_CPU.Text = Replace(TA_PC_CPU.Text, "(Tm)", "")
  TA_PC_CPU.Text = Replace(TA_PC_CPU.Text, "(TM)", "")
  TA_PC_CPU.Text = Replace(TA_PC_CPU.Text, "(tM)", "")
  If InStr(TA_PC_CPU.Text, "Pr") > 0 Then
    TA_PC_CPU.Text = Left(TA_PC_CPU.Text, InStr(TA_PC_CPU.Text, "Pr") - 2)
  End If
  TA_PC_CPU.Text = Trim(TA_PC_CPU.Text)
  ' Velocitat de la CPU en Ghz
  ' CPU Speed in Ghz
  Shell "lscpu | grep 'CPU max MHz'" To c_SysInfo3
  If Trim(c_SysInfo3) = "" Then
    Shell "lscpu | grep 'CPU MHz'" To c_SysInfo3
  End If
  ' 15 de novembre 2021
  If InStr(c_SysInfo3, ".") > 0 Then
    c_SysInfo3 = Replace(c_SysInfo3, ".", ",", gb.Binary)
  End If
  ' 15 de novembre 2021
  TB_PC_CPU_Ghz.Text = Trim(Str(Val(Mid(c_SysInfo3, 14, InStr(c_SysInfo3, ",") - 1)) / 1000))
  If Len(TB_PC_CPU_Ghz.Text) > 4 Then
    TB_PC_CPU_Ghz.Text = Left(TB_PC_CPU_Ghz.Text, 4)
  End If
  ' Total de GB del Disc Dur Principal
  ' Total amount of GB in the Main Hard Disk Drive
  Shell "lsblk -o 'NAME,SIZE' | grep 'nvme'" To c_SysInfo4
  n_inici_GB_HDD = 8
  If Trim(c_SysInfo4) = "" Then
    Shell "lsblk -o 'NAME,SIZE' | grep 'sd'" To c_SysInfo4
    n_inici_GB_HDD = 4
  Else If Trim(c_SysInfo4) = "" Then
    Shell "lsblk -o 'NAME,SIZE' | grep 'hd'" To c_SysInfo4
    n_inici_GB_HDD = 4
  End If
  If InStr(c_SysInfo4, ",") > 0 Then
    n_fi_GB_HDD = InStr(c_SysInfo4, ",") - (n_inici_GB_HDD)
  Else
    n_fi_GB_HDD = InStr(c_SysInfo4, "G") - (n_inici_GB_HDD)
  End If
  TB_PC_GB_HDD.Text = Trim(Mid(c_SysInfo4, n_inici_GB_HDD, n_fi_GB_HDD))
  ' Marca GPU - targeta gràfica
  ' GPU brand - Grpahic Processor Unit Brand
  Shell "lspci | grep ' VGA '" To c_SysInfo5
  If InStr(UCase(c_SysInfo5), "NVIDIA") > 0 Then
    c_GPU = "nVidia"
  Else If InStr(UCase(c_SysInfo5), "AMD ") > 0 Or InStr(UCase(c_SysInfo5), "ATI ") > 0 Or
  InStr(UCase(c_SysInfo5), " AMD") > 0 Or InStr(UCase(c_SysInfo5), " ATI") > 0 Then
    c_GPU = "AMD - ATI"
  Else If InStr(UCase(c_SysInfo5), "INTEL") > 0 Then
    c_GPU = "Intel"
  Else
    c_GPU = "Other GPUs"
  End If
  ' MB de la GPU - targeta gràfica
  ' GPU total of RAM in MB
  Shell "lspci -v | grep ' prefetchable'" To c_SysInfo6
  If Trim(c_SysInfo6) = "" Then
    Shell "lspci -v | grep 'Prefetchable memory behind bridge: 0" To c_SysInfo6
  End If
  TB_PC_GPU_MB.Text = Trim(Mid(c_SysInfo6, InStr(c_SysInfo6, "size=") + 5, 4))
  If InStr(TB_PC_GPU_MB.Text, "M") > 0 Then
    TB_PC_GPU_MB.Text = Trim(Left(TB_PC_GPU_MB.Text, InStr(TB_PC_GPU_MB.Text, "M") - 1))
  Else If InStr(TB_PC_GPU_MB.Text, "G") > 0 Then
    TB_PC_GPU_MB.Text = Trim(Str(Val(Left(TB_PC_GPU_MB.Text, InStr(TB_PC_GPU_MB.Text, "G") - 1)) * 1024))
  End If
  ' SSD - M.2 si o no
  ' SSD - M.2 yes or not
  Shell "cat /sys/block/sda/queue/rotational" To c_SysInfo7
  If Trim(c_SysInfo7) = "0" Then
    bSSD = True
  Else
    Shell "lsblk -d -o name,rota | grep 'nvme'" To c_SysInfo7
    If InStr(c_SysInfo7, "0") > 0 Then
      bSSD = True
    Else
      bSSD = False
    End If
  End If
  Shell "lsb_release -d" To c_SysInfo8
  TA_PC_OS.Text = Trim(Mid(c_SysInfo8, 14, Len(Trim(c_SysInfo8)) - 13))
  ' Sistema Operatiu de 32 o de 64 bits
  ' 32 or 64 bits Operating System
  Shell "uname -m" To c_SysInfo9
  If InStr(Trim(c_SysInfo9), "64") > 0 Then
    c_SysInfo9 = "64 bits"
  Else
    c_SysInfo9 = "32 bits"
  End If
  TA_PC_OS2.Text = c_SysInfo9
  ' Total de nuclis del processador
  ' CPU total of cores
  Shell "cat /proc/cpuinfo | grep 'cpu cores'" To c_SysInfo10
  c_CPU = Trim(Mid(c_SysInfo10, InStr(c_SysInfo10, ":") + 1, 2))
  n_CPU_cores = Val(c_CPU)
  If n_CPU_cores > 1 Then
    c_CPU = " Cores"
  Else 
    c_CPU = " Core"
  End If
  ' Total de fils del processador
  ' CPU Total of threads
  Shell "lscpu | grep 'CPU(s):'" To c_SysInfo10
  c_Threads = Trim(Mid(c_SysInfo10, InStr(c_SysInfo10, "N") - 2, 2))
  ' Thanks to Brian G. for this code, for all Tria S.O. 2.0.x
  Try n_CPU_Threads = (Val(c_Threads) * n_CPU_cores)
  If Error Then
    c_Threads = Trim(Mid(c_SysInfo10, 8, 28))
    n_CPU_Threads = (Val(c_Threads) * n_CPU_cores)
  End If
  ' Thanks to Brian G. for this code, for all Tria S.O. 2.0.x
  If n_CPU_Threads > 1 Then
    c_Threads = " Threads"
  Else 
    c_Threads = " Thread"
  End If
  TA_PC_OS2.Text = TA_PC_OS2.Text & "; " & Trim(Str(n_CPU_cores)) & c_CPU & ", " & Trim(Str(n_CPU_Threads)) & c_Threads
  ' TPM = Trusted Platform Module
  Shell "dmesg | grep -i tpm" To c_SysInfo11
  If InStr(c_SysInfo11, "No TPM chip found", 1) > 0 Then
    TB_TPM.Text = "NO"
    nTPM = 0
  Else If InStr(c_SysInfo11, " 1.0", 1) > 0 Then
    TB_TPM.Text = "1.0"
    nTPM = 1.0
  Else If InStr(c_SysInfo11, " 1.2", 1) > 0 Then
    TB_TPM.Text = "1.2"
    nTPM = 1.2
  Else If InStr(c_SysInfo11, "TPM2", 1) > 0 Then
    TB_TPM.Text = "2.0"
    nTPM = 2.0
  End If
  ' Direct X, WDDM & OpenGL version
  ' November 2021
  ' Thanks to Brian G. for this code, for all Tria S.O. 2.0.x
  Try Shell "glxinfo | grep 'OpenGL version'" To c_SysInfo12
  If Error Then 
    TB_OpenGL.Text = "4.6"
    c_OpenGL_ver = "4.6"
  Else 
    TB_OpenGL.Text = Trim(Mid(c_SysInfo12, 23, 4))
    c_OpenGL_ver = TB_OpenGL.Text
  End If
  Try n_OpenGL_ver = Val(c_OpenGL_ver)
  If Error Then
    c_OpenGL_ver = Replace(TB_OpenGL.Text, ".", ",", gb.Binary)
    n_OpenGL_ver = Val(c_OpenGL_ver)
  End If
  ' November 2021
  ' Thanks to Brian G. for this code, for all Tria S.O. 2.0.x
  Select Case n_OpenGL_ver
  Case 1.0
    TB_DirectX_version.Text = "3"
    TB_WDDM.Text = "0.0"
  Case 1.1
    TB_DirectX_version.Text = "5"
    TB_WDDM.Text = "0.0"
  Case 1.3
    TB_DirectX_version.Text = "8"
    TB_WDDM.Text = "0.0"
  Case 1.4 To 2.0
    TB_DirectX_version.Text = "9"
    TB_WDDM.Text = "0.0"
  Case 2.1 To 3.1
    TB_DirectX_version.Text = "10"
    TB_WDDM.Text = "1.0"
  Case 3.2 To 4.4
    TB_DirectX_version.Text = "11"
    If n_OpenGL_ver = 3.2 Then
      TB_WDDM.Text = "1.1"
    Else If n_OpenGL_ver = 3.3 Then
      TB_WDDM.Text = "1.2"
    Else 
      TB_WDDM.Text = "1.3"
    End If
  Case 4.5 To 4.6
    TB_DirectX_version.Text = "12"
    If n_OpenGL_ver = 4.5 Then
      TB_WDDM.Text = "2.0"
    Else 
      TB_WDDM.Text = "2.1"
    End If
  End Select
  ' Detectar grabadora òptica (CD, DVD o Blu-Ray)
  ' Detects Optical Drive (CD, DVD or Blu-Ray)
  Shell "lshw -short | grep 'D-RW'" To c_SysInfo13
  If InStr(c_SysInfo13, "D-RW", 1) > 0 Then 
    b_Optical_Drive = True
  Else 
    b_Optical_Drive = False
  End If
  ' Detectar internet
  ' Detects internet
  Try Shell "ping -q -c 1 -W 1 google.com | grep 'transmitted'" To c_SysInfo14
  If Error Then
    c_SysInfo14 = "1 packets transmitted, 1 received, 0% packet loss, time 0ms"
  End If
  If InStr(c_SysInfo14, "1 received", 1) > 0 Then 
    b_internet = True
  Else
    b_internet = False
  End If
  ' Detectar WiFi
  ' Detects WiFi
  Shell "lshw -C network | grep 'description'" To c_SysInfo15
  If InStr(c_SysInfo15, "Wireless", 1) > 0 Then
    b_Wifi = True
  Else 
    b_Wifi = False
  End If
  ' Detectar si es ordinador portàtil
  ' Detects if the PC is a laptop
  Shell "cat /sys/class/dmi/id/chassis_type" To c_SysInfo16
  nLaptop = Val(Trim(c_SysInfo16))
  If (nLaptop > 7 And nLaptop < 11) Or (nLaptop = 14) Then 
    b_Laptop = True
  Else 
    b_Laptop = False
  End If
  ' Textes en el idioma seleccionat al inici
  ' Texts in the language selected at the beginning
  Select Case Lbl_Lang.Text
  Case "русский язык"
      Label1.Text = "Выберите операционную систему - версия 2.0.3 (c) 2022, автор: Джоан Баптиста ЭЛИС Джонни Б. Гуд R'n'R"
      Label2.Text = "Это приложение поможет вам выбрать лучшую операционную систему для любого компьютера, выбрав основные характеристики любого ПК."
      Label13.Text = "TriaSO старается помочь как можно большему количеству людей, в том числе и тем, у кого есть только компьютер 25-летней давности."
      ChB_Min_Win.Text = "Отметьте красным то, чего не хватает, чтобы добраться:"
      TS_Min_Win.Text = "Отметьте красным то, чего не хватает, чтобы добраться:"

    NumericUpDownCPU_Hz.Step = 33
    NumericUpDownCPU_Hz.MinValue = 800
    NumericUpDownCPU_Hz.MaxValue = 999

    NumericUpDownCPU_Hz.MaxValue = 5
    NumericUpDownCPU_Hz.MinValue = 1

    NumericUpDownCPU_Hz.MaxValue = 999
    NumericUpDownCPU_Hz.MinValue = 66
    NumericUpDownCPU_Hz.Step = 33

    NumericUpDownCPU_Hz.MaxValue = 5
    NumericUpDownCPU_Hz.MinValue = 1
    NumericUpDownCPU_Hz.Step = 1
  End If
End

Public Sub Bt_Calcul_Result_Click()
  ' Càlcul de punts assignats depenent dels diferents components del PC
  ' Calculation of assigned points depending on the different components of the PC
  If ChB_OnlyFREE_OS.Value = False And ChB_OnlyWin.Value = False Then
    ' Missatge demanant que es trii algun tipus de Sistema Operatiu
    ' en cas que s'hagin desmarcat els dos tipus
    ' Message requesting that some type of Operating System be chosen
    ' in case both types have been unchecked
    Select Case Lbl_Lang.Text
    Case "Català"
      Message.Info("Tria al menys un tipus de Sistema Operatiu a la pestanya <System> i torna a polsar el botó <Calcular Resultat>.")
    Case "Deutsche"
      Message.Info("Wählen Sie auf der Registerkarte <System> mindestens einen Betriebssystemtyp aus und drücken Sie dann erneut auf die Schaltfläche <Ergebnis berechnen>.")
    Case "English"
      Message.Info("Wählen Sie auf der Registerkarte <System> mindestens einen Betriebssystemtyp aus und drücken Sie dann erneut die Schaltfläche <Ergebnis berechnen>.")
    Case "Español"
      Message.Info("Escoger mínimo un tipo de Sistema Operativo en la pestaña <System> y volver a pulsar el botón <Calcular Resultado>.")
    Case "Euskara"
      Message.Info("Hautatu gutxienez Sistema Eragile mota bat <System> fitxan eta sakatu berriro <Kalkulatu Emaitza> botoia.")
    Case "Française"
      Message.Info("Choisissez au moins un type de système d'exploitation dans l'onglet <System> et appuyez à nouveau sur le bouton <Calculer le Résultat>.")
    Case "Galego"
      Message.Info("Escolha pelo menos um tipo de sistema operacional na guia <System> e pressione o botão <Calcular resultado> novamente.")
    Case "Indonesia"
      Message.Info("Pilih setidaknya satu jenis Sistem Operasi di tab <System> lalu tekan lagi tombol <Calculate Result>.")
    Case "Italiano"
      Message.Info("Scegliere almeno un tipo di sistema operativo nella scheda <System> e premere nuovamente il pulsante <Calcola Risultato>.")
    Case "Nederlands"
      Message.Info("Selecteer ten minste één type besturingssysteem op het tabblad <System> en druk vervolgens opnieuw op de knop <Resultaat berekenen>.")
    Case "Português"
      Message.Info("Escolha pelo menos um tipo de sistema operacional na guia <System> e pressione o botão <Calcular Resultado> novamente.")
    Case "русский язык"
      Message.Info("Выберите хотя бы один тип операционной системы на вкладке <System>, а затем снова нажмите кнопку <Рассчитать результат>.")
    Case "Română"
      Message.Info("Selectați cel puțin un tip de sistem de operare în fila <System> și apoi apăsați din nou butonul <Calculare rezultat>.")    
    Case "Suomi"
      Message.Info("Valitse vähintään yksi käyttöjärjestelmän tyyppi <System> -välilehdestä ja paina <Laske tulos> -painiketta uudelleen.")
    End Select
  Else
    If ChB_OnlyFREE_OS.Value = False Then
      ' Desactivo mostrar resultat Linux quan el usuari no el vol
      ' Hide Linux results when the user request it
      TL_Linux.Visible = False
      LinkLinux.Visible = False
    Else
      TL_Linux.Visible = True
      LinkLinux.Visible = True
    End If
    If ChB_OnlyWin.Value = False Then
      ' Desactivo mostrar resultat Windows quan el usuari no el vol
      ' Hide Windows results when the user request it
      TL_Windows.Visible = False
      LinkWindows.Visible = False
    Else
      TL_Windows.Visible = True
      LinkWindows.Visible = True
    End If
    TS_Resultat.Visible = True
    ' Càlcul de punts assignats depenent dels diferents components del PC
    ' Calculation of assigned points depending on the different components of the PC
    ' CPU
    If ChB_Mhz.Value = True Then
      nCPU = (NumericUpDownCPU_Hz.Value * 0.3)
    Else
      nCPU = (((NumericUpDownCPU_Hz.Value * 100) + NumericUpDownMhz.Value) * 75) / 100
    End If
    If ChB_CPU_x64.Value = True Then
    ' a les CPU de 64 bits li sumem 49 punts
    ' 64 bit CPU's = add 49 points
      nCPU = nCPU + 49
    End If
    If ChB_CPU_Multi_Core.Value = True Then
    ' a les CPU de 2 o mes nuclis li sumem 25 punts per cada FIL
    ' 25 extra points for each thread in multi-core CPUs
      If NumericUpDown_CPU_Threads.Value > 1 Then
        nCPU = nCPU + (NumericUpDown_CPU_Threads.Value * 25)
      End If
    End If
    ' RAM
    If ChB_RAM_MB.Value = True Then
      nRAM = (NumericUpDownRAM.Value / 32) * 3
    Else
      nTemp = (NumericUpDownRAM.Value * 1024) + (NumericUpDownRAM_MB.Value * 10)
      nRAM = (nTemp / 32) * 2
      nRAM = nRAM + 44
    End If
    ' GPU - targeta gràfica
    nTemp = (NumericUpDown_GPU_MB_RAM.Value / 32)
    If ChB_Embeded_GPU.Value = True Then
      If RB_intel.Value = True Then
        nGPU = (nTemp * 1.5) + 24
      Else
        nGPU = (nTemp * 1.5) + 12
      End If
    Else
      If RB_AMD_ATI.Value = True Then
        nGPU = (nTemp * 2)
      Else
        nGPU = (nTemp * 2.5)
      End If
    End If
    ' Disc Dur - Hard Disk Drive
    nHDD = NumericUpDownHDD_GB.Value - 10
    ' Si es un disc SSD o M.2 se li assignen 90 punts extra
    ' If SSD or M.2 we add 90 extra points
    If ChB_SSD.Value = True Then
      nHDD = nHDD + 90
    End If
    ' Calcul de puntuació total, incloent decimals de RAM i CPU
    ' Total amount of points = CPU + RAM + GPU + HDD
    nPuntuacio_Total = nCPU + nRAM + nGPU + nHDD
    If ChB_NO_internet.Value = True Then
      If nPuntuacio_Total > 699 Then
        nPuntuacio_Total = nPuntuacio_Total - 200
      Else If nPuntuacio_Total > 499 Then
        nPuntuacio_Total = nPuntuacio_Total - 125
      Else If nPuntuacio_Total > 249 Then
        nPuntuacio_Total = nPuntuacio_Total - 75
      Else
         nPuntuacio_Total = nPuntuacio_Total - 25
      End If
    End If
    ' Si es un portàtil amb sistema operatiu bastant nou això es bo i dona punts,
    ' ja que serà factible fer ampliacions de maquinari (RAM - disc dur) per
    ' passar a un sistema operatiu mes nou (o mes bo) i si es pc portàtil però te
    ' un S.O. antic, serà mes difícil que se li pugui ampliar el hardware (incloent
    ' targeta gràfica quan parlem de Windows) per poder passar a un S.O. mes modern
    
    ' If is a laptop with a quite new operating system this is good and adds points,
    ' because it will be possible to update the hardware (RAM - Hard Disk) to be
    ' able to upgrade to a newer (or a better) operating system and if is a laptop
    ' but with an old O.S., it will be more difficult to upgrade that hardware
    ' (including graphic card in Windows cases) to upgrade to a modern O.S
    If ChB_Laptop.Value = True Then
      If nPuntuacio_Total < 500 Then
         nPuntuacio_Total = nPuntuacio_Total - 10
      Else
         nPuntuacio_Total = nPuntuacio_Total + 5
      End If
      ' I si es connecta per wifi, millorem poc la nota només en sistemes operatius
      ' moderns, en ordinadors que tenen prou puntuació com permetre wifis prou ràpids
      ' com per permetre actualitzacions SOVINT via wifi (en Windows, clarament CAP
      ' anterior a Windows 7, segons la meva experiència, quan parlem de wifis antics)
      
      ' And if connected via wifi, we add a few points only in modern operating systems,
      ' in computers that have points enough to handle fast wifi connections, good enough
      ' to allow COMMON updates via wifi (on Windows, clearly NONE vefore Windows 7, as
      ' i have experienced, related with older wifis)
      If ChB_Wifi.Value = True Then
        If nPuntuacio_Total > 499 Then
          nPuntuacio_Total = nPuntuacio_Total + 5
        End If
      End If
    End If
    ' Requisits RECOMANABLES per Windows 11 (que no MÍNIMS):
    ' Windows 11 hardware RECOMENDATIONS (not minimal, nor system requirements).
    ' En casos que s'hagi marcat una CPU, GPU i/o quantitat RAM "low" i el programa recomani
    ' un Sistema Operatiu massa nou, aquí es on el programa fa la seva millor feina.
    ' When the total amount of points is recomending a too new Operating System just because
    ' one or more pieces of that hardware, here is when this app does their best job
    If nPuntuacio_Total > 999 And ChB_CPU_x64.Value = False Then
      ' Si el processador no es de 64 bits, no es pot fer servir Windows 11 i els baixa a grup Windows 10
      ' If the processor is not 64-bit, you cannot use Windows 11 and download them to Windows 10 Group
      nPuntuacio_Total = 999
      If ChB_Min_Win.Value = True And RB_Win11.Value = True Then
        ChB_CPU_x64.Foreground = Color.RGB(255, 0, 0)
      Else
        ChB_CPU_x64.Foreground = Color.RGB(0, 0, 0)
      End If
    Else If (nPuntuacio_Total > 999 And ChB_CPU_x64.Value = True And
    ChB_Min_Win.Value = True And RB_Win11.Value = True) Then
        ChB_CPU_x64.Foreground = Color.RGB(0, 0, 0)
    End If
    If nPuntuacio_Total > 999 And ChB_SSD.Value = False Then
      ' Si no hi ha disc SSD (normal, M.2 o U.2) no es pot fer servir Windows 11 i els baixa a grup Windows 10
      ' If there is no SSD (normal, M.2 or U.2) you cannot use Windows 11 and download them to Windows 10 group
      nPuntuacio_Total = 999
      If ChB_Min_Win.Value = True And RB_Win11.Value = True Then
        ChB_SSD.Foreground = Color.RGB(255, 0, 0)
      Else
        ChB_SSD.Foreground = Color.RGB(0, 0, 0)
      End If
    Else If (nPuntuacio_Total > 999 And ChB_SSD.Value = True And
    ChB_Min_Win.Value = True And RB_Win11.Value = True) Then
        ChB_SSD.Foreground = Color.RGB(0, 0, 0)
    End If
    If nPuntuacio_Total > 999 And ChB_CPU_Multi_Core.Value = False Then
      ' Processadors que no tinguin 2 o mes nuclis no es pot fer servir Windows 11 i els baixa a grup Windows 10
      ' Processors that do not have 2 or more cores cannot use Windows 11 and download them to the Windows 10 group
      nPuntuacio_Total = 999
      If ChB_Min_Win.Value = True And RB_Win11.Value = True Then
        ChB_CPU_Multi_Core.Foreground = Color.RGB(255, 0, 0)
      Else
        ChB_CPU_Multi_Core.Foreground = Color.RGB(0, 0, 0)
      End If
    Else If (nPuntuacio_Total > 999 And ChB_CPU_Multi_Core.Value = True And
    ChB_Min_Win.Value = True And RB_Win11.Value = True) Then
        ChB_CPU_Multi_Core.Foreground = Color.RGB(0, 0, 0)
    End If
    If nPuntuacio_Total > 999 And ((ChB_Mhz.Value = False And NumericUpDownCPU_Hz.Value < 2) Or (ChB_Mhz.Value = True)) Then
      ' Processadors que no tinguin 2 o mes Ghz no es pot recomanar Windows 11 i els baixa a grup Windows 10
      ' Processors that do not have 2 or more Ghz cannot be recommended for Windows 11 and are downgraded to Windows 10 group
      nPuntuacio_Total = 999
      If ChB_Min_Win.Value = True And RB_Win11.Value = True Then
        NumericUpDownCPU_Hz.Foreground = Color.RGB(255, 0, 0)
        NumericUpDownMhz.Foreground = Color.RGB(255, 0, 0)
      Else
        NumericUpDownCPU_Hz.Foreground = Color.RGB(0, 0, 0)
        NumericUpDownMhz.Foreground = Color.RGB(0, 0, 0)
      End If
    Else If (nPuntuacio_Total > 999 And ChB_Min_Win.Value = True And RB_Win11.Value = True And ChB_Mhz.Value = False And NumericUpDownCPU_Hz.Value > 1) Then
      NumericUpDownCPU_Hz.Foreground = Color.RGB(0, 0, 0)

        NumericUpDownRAM_MB.Foreground = Color.RGB(255, 0, 0)
      Else

      NumericUpDownRAM.Foreground = Color.RGB(0, 0, 0)
      NumericUpDownRAM_MB.Foreground = Color.RGB(0, 0, 0)
    End If
    If nPuntuacio_Total > 999 And NumericUpDownDirectX.Value < 12 Then
      ' Si DirectX es versió anterior a la 12 no es pot recomanar Windows 11 i els baixa al grup Windows 10

      Else
        NumericUpDownDirectX.Foreground = Color.RGB(0, 0, 0)
      End If
    Else If (nPuntuacio_Total > 999 And NumericUpDownDirectX.Value > 11 And

        NumUpDown_WDDM_ver.Foreground = Color.RGB(255, 0, 0)
        NumUpDown_WDDM_ver2.Foreground = Color.RGB(255, 0, 0)

        NumUpDown_WDDM_ver2.Foreground = Color.RGB(0, 0, 0)
      End If

    If nPuntuacio_Total > 999 And (RB_TPM_2_0.Value = False Or ChB_TPM.Value = False) Then

Public Sub Bt_charge_PC_Click()
  Dim n_Mhz As Float
  Dim n_MB As Float
  Dim n_Ghz As Float
  Dim n_GB As Float
  Dim n_GPU As Integer
  If InStr(TA_PC_OS2.Text, "64 bits") Then
    ChB_CPU_x64.Value = True
  Else 
    ChB_CPU_x64.Value = False
  End If
  If nRAM_MB > 1023 Then
    ChB_RAM_MB.Value = False
    NumericUpDownRAM.MaxValue = 96
    NumericUpDownRAM.MinValue = 1
    NumericUpDownRAM.Step = 1
    If nRAM_MB > 31743 Then
      nRAM_MB = 31744
    Endif
    NumericUpDownRAM.Value = nRAM_MB / 1024
  Else If ChB_RAM_MB.Value = True And nRAM_MB < 2049 Then
    NumericUpDownRAM.Value = nRAM_MB
  Else
    ChB_RAM_MB.Value = False
    NumericUpDownRAM.MaxValue = 96
    NumericUpDownRAM.MinValue = 1
    NumericUpDownRAM.Step = 1
    NumericUpDownRAM.Value = nRAM_MB / 1024
  End If
  If Val(TB_PC_RAM.Text) > 1125 Then
    n_MB = Val(TB_PC_RAM.Text) - (NumericUpDownRAM.Value * 1024)
    NumericUpDownRAM_MB.Value = Int(n_MB / 10)
  Else
    NumericUpDownRAM_MB.Value = 0
  End If
  If InStr(UCase(TA_PC_CPU.Text), "CORE") > 0 Then
    ChB_CPU_x64.Value = True
    ChB_CPU_Multi_Core.Value = True
    ChB_Mhz.Value = False
    Lbl_Mhz_Ghz.Text = "GigaHertzs"
  End If
  If n_CPU_cores > 1 Then
    ChB_CPU_Multi_Core.Value = True
  Endif
  NumericUpDownCPU_cores.Value = n_CPU_cores
  If ChB_CPU_Multi_Core.Value = True Then
    NumericUpDownCPU_cores.Enabled = True
    NumericUpDown_CPU_Threads.Value = n_CPU_Threads
    If n_CPU_Threads = (n_CPU_cores * 2) Then
      ChB_HT.Value = True
    Else
      ChB_HT.Value = False
    Endif
  Else
    NumericUpDownCPU_cores.Enabled = False
    NumericUpDownCPU_cores.Value = 1
    NumericUpDown_CPU_Threads.Value = 1
  Endif
  n_Ghz = Val(TB_PC_CPU_Ghz.Text)
  NumericUpDownCPU_Hz.Value = Int(n_Ghz)
  If n_Ghz > 1.09 Then
    n_Mhz = (n_Ghz * 1000) - ((NumericUpDownCPU_Hz.Value) * 1000)
    NumericUpDownMHz.Value = Int(n_Mhz / 10)
  Else
    NumericUpDownMHz.Value = 0
  End If
  n_GB = Val(TB_PC_GB_HDD.Text)
  If n_GB > 990 Then
    NumericUpDownHDD_GB.Value = 990
  Else
    NumericUpDownHDD_GB.Value = n_GB
  End If
  n_GPU = Val(TB_PC_GPU_MB.Text)
  If n_GPU > 8192 Then
    NumericUpDown_GPU_MB_RAM.Value = 8192
  Else If n_GPU < 32 Then
    NumericUpDown_GPU_MB_RAM.Value = 32
  Else
    NumericUpDown_GPU_MB_RAM.Value = n_GPU
  End If
  'If NumericUpDown_GPU_MB_RAM.Value < 128 Then
  '  NumericUpDownDirectX.Value = 6
  'Else If NumericUpDown_GPU_MB_RAM.Value < 256 And NumericUpDown_GPU_MB_RAM.Value > 127 Then
  '  NumericUpDownDirectX.Value = 8
  'Else If NumericUpDown_GPU_MB_RAM.Value = 256 Then
  '  NumericUpDownDirectX.Value = 9
  'Else If NumericUpDown_GPU_MB_RAM.Value > 256 And NumericUpDown_GPU_MB_RAM.Value < 513 Then
  '  NumericUpDownDirectX.Value = 10
  'Else If NumericUpDown_GPU_MB_RAM.Value > 512 Then
  '  NumericUpDownDirectX.Value = 12
  'End If
  Select Case c_GPU
    Case "nVidia"
      ChB_Embeded_GPU.Value = False
      RB_nVidia.Enabled = True
      RB_AMD_ATI.Enabled = True
      RB_nVidia.Value = True
      RB_AMD_ATI.Value = False
      RB_intel.Value = False
      RB_intel.Enabled = False
      RB_other_GPUs.Value = False
      RB_other_GPUs.Value = False
    Case "AMD - ATI"
      ChB_Embeded_GPU.Value = False
      RB_nVidia.Enabled = True
      RB_AMD_ATI.Enabled = True
      RB_nVidia.Value = False
      RB_AMD_ATI.Value = True
      RB_intel.Value = False
      RB_intel.Enabled = False
      RB_other_GPUs.Value = False
      RB_other_GPUs.Value = False
    Case "Intel"
      ChB_Embeded_GPU.Value = True
      RB_nVidia.Enabled = False
      RB_AMD_ATI.Enabled = False
      RB_nVidia.Value = False
      RB_AMD_ATI.Value = False
      RB_intel.Value = True
      RB_intel.Enabled = True
      RB_other_GPUs.Enabled = True
      RB_other_GPUs.Value = False
    Case "Other GPUs"
      ChB_Embeded_GPU.Value = True
      RB_nVidia.Enabled = False
      RB_AMD_ATI.Enabled = False
      RB_nVidia.Value = False
      RB_AMD_ATI.Value = False
      RB_intel.Value = False
      RB_intel.Enabled = True
      RB_other_GPUs.Enabled = True
      RB_other_GPUs.Value = True
  End Select
  If bSSD = True Then
    ChB_SSD.Value = True
  Else
    ChB_SSD.Value = False
  End If  
  If NumericUpDown_GPU_MB_RAM.Value < 512 Or NumericUpDownDirectX.Value < 11 Then
    RB_DM_High.Value = False
    RB_DM_High.Enabled = False
  End If
  If NumericUpDown_GPU_MB_RAM.Value < 256 Or NumericUpDownDirectX.Value < 9 Then
    RB_DM_Medium.Value = False
    RB_DM_Medium.Enabled = False
    RB_DM_Light.Value = True
  End If
  If TB_TPM.Text = "NO" Then
    ChB_TPM.Value = False
    RB_TPM_1_0.Enabled = False
    RB_TPM_1_2.Enabled = False
    RB_TPM_2_0.Enabled = False
  Else 
    ChB_TPM.Value = True
    RB_TPM_1_0.Enabled = True
    RB_TPM_1_2.Enabled = True
    RB_TPM_2_0.Enabled = True
    If TB_TPM.Text = "1.0" Then
      RB_TPM_1_0.Value = True
    Else If TB_TPM.Text = "1.2" Then
      RB_TPM_1_2.Value = True
    Else
      RB_TPM_2_0.Value = True
    End If
  End If
  Select Case n_OpenGL_ver
  Case 0 To 1.2
    NumericUpDownDirectX.Value = 7
    NumUpDown_WDDM_ver.Value = 0
    NumUpDown_WDDM_ver2.Value = 0
  Case 1.3
    NumericUpDownDirectX.Value = 8
    NumUpDown_WDDM_ver.Value = 0
    NumUpDown_WDDM_ver2.Value = 0
  Case 1.4 To 2.0
    NumericUpDownDirectX.Value = 9
    NumUpDown_WDDM_ver.Value = 0
    NumUpDown_WDDM_ver2.Value = 0
  Case 2.1 To 3.1
    NumericUpDownDirectX.Value = 10
    NumUpDown_WDDM_ver.Value = 1
    NumUpDown_WDDM_ver2.Value = 0
  Case 3.2 To 4.4
    NumericUpDownDirectX.Value = 11
    NumUpDown_WDDM_ver.Value = 1
    If n_OpenGL_ver = 3.2 Then
      NumUpDown_WDDM_ver2.Value = 1
    Else If n_OpenGL_ver = 3.3 Then
      NumUpDown_WDDM_ver2.Value = 2
    Else 
      NumUpDown_WDDM_ver2.Value = 3
    End If
  Case 4.5 To 4.6
    NumericUpDownDirectX.Value = 12
    NumUpDown_WDDM_ver.Value = 2
    NumUpDown_WDDM_ver2.Value = 0
  End Select
  ' Detectar grabadora òptica (CD, DVD o Blu-Ray)
  ' Detects Optical Drive (CD, DVD or Blu-Ray)
  If b_Optical_Drive = False Then
    ChB_No_Optical_Drive.Value = True
  Else 
    ChB_No_Optical_Drive.Value = False
  End If
  ' Detectar internet
  ' Detects internet
  If b_internet = True Then
    ChB_NO_internet.Value = False
  Else
    ChB_NO_internet.Value = True
  End If
  ' Detectar Wifi
  ' Detects Wifi
  If b_Wifi = True Then 
    ChB_Wifi.Value = True
  Else 
    ChB_Wifi.Value = False
  End If
  ' Detectar si es ordinador portàtil
  ' Detects if the PC is a laptop
  If b_Laptop = True Then 
    ChB_Laptop.Value = True
  Else 
    ChB_Laptop.Value = False
  End If
End
